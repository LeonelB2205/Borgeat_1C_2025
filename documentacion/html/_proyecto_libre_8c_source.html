103static void controlLed(void *vparametro)
104{
105 while (true)
106 {
107 if (inicio)
108 {
109 if (encendido == false)
110 {
111 if (iterador == LENGTH_SECUENCIA)
112 {
113 GPIOOff(misLeds[ledActual].led);
114 inicio = false;
115 tiempoRespuestaTotal = tiempoRespuestaTotal/LENGTH_SECUENCIA;
116 printf("%d\r\n", tiempoRespuestaTotal);
117 salidaBluetooth();
118 }
119 else
120 {
121 if (iterador > 0)
122 GPIOOff(misLeds[ledActual].led);
123
124 ledActual = secuencia[iterador];
125 GPIOOn(misLeds[ledActual].led);
126 encendido = true;
127 iterador++;
128 }
129 }
130 }
131 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
132 }

138static void medir(void *vparametro)
139{
140 uint16_t contador = 0;
141 while (true)
142 {
143 if (inicio)
144 {
145 if (encendido)
146 {
147 if (!GPIORead(misLeds[ledActual].tecla))
148 {
149 tiempoRespuesta = (contador * CONFIG_TECLAS_us) / 1000;
150 tiempoRespuestaTotal = tiempoRespuestaTotal + tiempoRespuesta;
151 encendido = false;
152 contador = 0;
153 }
154 else
155 {
156 contador++;
157 }
158 }
159 }
160 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
161 }
